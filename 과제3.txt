# 소수를 구하는 루틴을 yield generator를 사용하여 구현.

def gen_primes(N):
    """N까지의 소수를 생성"""
    primes = set()
    for n in range(2, N):
        if all(n % p > 0 for p in primes):
            primes.add(n)
    return primes

p = gen_primes(10)
print(p)

def gen_primes(N):
    primes = set()
    for n in range(2, N):
        if all(n % p > 0 for p in primes):
            primes.add(n)
            yield n

print(*gen_primes(10))
